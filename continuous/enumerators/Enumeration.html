A operator [](int i) #

A operator [](int i) {
  var ps = parts;
  var it = i;
  while (true) {
    if (ps.isEmpty()) throw "index out of range";
    if (it < ps.head.card) return ps.head[it];
    it = it - ps.head.card;
    ps = ps.tail;
  }
}
